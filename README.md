本题正解：

```c
#define MAX(a, b)   (((a) > (b)) ? (a) : (b))

int MaxSubArray(const int nums[], int numSize)
{
    int sum = 0, res = nums[0];
    for(int i = 0; i < numSize; i++)
    {
        const int n = nums[i];
        sum = MAX(sum + n, n);
        res = MAX(res, sum);
    }
    
    return res;
}

```
